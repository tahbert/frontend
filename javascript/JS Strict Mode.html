Strict Mode

Strict Mode

Elements Explaination Example
Useful
  • throw an error
  • make it easier to write "secure" JS
  • change previously accepted "bad syntax" into real errors
  • make it impossible to accidentally create a global variable
Declaring "use strict";
  • at the beginning of script => global scope
  • at the beginning of func => local scope
 "use strict";
x = 2; // error
myFn();
function myFn() {y = 2;} // error
x = 2; // not throw an error
myFn();
function myFn() {
 "use strict";
 y = 2; // error
}
impossible in strict mode
use a variable, object without declaring "use strict";
x = 2; // error
x = {fname:"Tahbert", lname:"Nguyen"}; // error
delete a variable, object, or function "use strict";
let x = 2;
function myFn() {};
delete x; // error
delete myFn; // error
duplicate a param "use strict";
function myFn(p,p) {}; // error
octal numeric literals "use strict";
let x = 010; // error
octal escape characters "use strict";
let x = "\010"; // error
write to a read-only property "use strict";
const obj = {};
Object.defineProperty(obj, "x", {value:0, writable:false});
obj.x = 2; // error
write to a get-only property "use strict";
const obj = {get x() {return 0} };
obj.x = 2; // error
delete an undeletable property "use strict";
delete Object.prototype; // error
eval, arguments words cannot be used as a variable "use strict";
let eval = 2; // error
with statement "use strict";
with (Math){x = cos(2)}; // error
eval() is not allowed to create variables in the scope from which it was called: "use strict";
eval ("let x = 2");
alert (x); // error
this keyword when the object is not specified
  • non-strict mode: func refers to the window object
  • strict mode: func refers to undefined
 "use strict";
function myFn() {
 alert(this); // return undefined
}
myFn();
