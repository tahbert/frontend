Meta Character (character with a special meaning)
\w match word characters
a-z, A-Z, 0-9, including underscore (_)
let pattern = "Give 100%!";
pattern.match(/\w/g);
\W match non-word characters let pattern = "Give 100%!";
pattern.match(/\W/g);
\d match a digit character let pattern = "Give 100%!";
pattern.match(/\d/g);
\D match a non-digit character let pattern = "Give 100%!";
pattern.match(/\D/g);
\s match a whitespace character let pattern = "This is Tahbert Nguyen";
pattern.match(/\s/g);
\S match a non-whitespace character let pattern = "This is Tahbert Nguyen";
pattern.match(/\S/g);
\b match at the beginning or end of a word let pattern = "Is this all there is?";

pattern.match(/is\b/gi);

\B match NOT at the beginning or end of a word let pattern = "Is this all there is?";

match "is" NOT at the beginning of a word

pattern.match(/\Bis/gi);

match "is" NOT at the end of a word

pattern.match(/is\B/gi);

match "is" NOT at the beginning and end of a word

Quantifier (define quantities)
+ n+ match any string that contains at least 1 n let pattern = "Hellooo World! Hello W3Schools!";

* n* match any string that contains 0 or more occurrences of n let pattern = "Hellooo World! Hello W3Schools!";

Range
[abc]
  • [abc] any of the characters a, b, or c
  • [A-Z] any character from uppercase A to uppercase Z
  • [a-z] any character from lowercase a to lowercase z
  • [A-z] any character from uppercase A to lowercase z
 let pattern = "Is this all there is?";

global search for "h" or "s"

pattern.match(/[hs]/g);

global search for character span [a-h]

pattern.match(/[a-h]/g);
[^abc]
  • [^abc] NOT any of the characters a, b, or c
  • [^A-Z] NOT any character from uppercase A to uppercase Z
  • [^a-z] NOT any character from lowercase a to lowercase z
  • [^A-z] NOT any character from uppercase A to lowercase z
 let pattern = "Is this all there is?";

global search for NOT "h" or "s"

pattern.match(/[^hs]/g);

global search for NOT character span [a-h]

pattern.match(/[^a-h]/g);
[0-9] match any digit between the brackets

global search for digit span [1-4]

let pattern = "123456789";
pattern.match(/[1-4]/g);
^[0-9] match any digit NOT between the brackets

global search for digit NOT in span [1-4]

let pattern = "123456789";
pattern.match(/[^1-4]/g);
Group
(x|y) match any of alternatives specified
() can contain []

global search for any of alternatives specified

let pattern = "re, green, red, green, gren, gr, blue, yellow";
pattern.match(/(red|green|blue)/g);
