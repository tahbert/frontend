Arithmetic Perform arithmetic on numbers
(the numbers in an arithmetic operation are called operands)
Arithmetic Operator Example
Addition + let x = 100 + 50; // literal
Subtraction - let x = a - b; // variable
Multiplication * let x = (100 + 50) * a; // expression
Division / let x = (100 + 50) / a; // expression
Modulus % let x = 5; let y = 2;
let z = x % y; // result: 1
Increment ++ let x = 5;
x++; // x = 6
Decrement -- let x = 5;
x--; // x = 4
Exponentiation ** let x = 5;
let z = x ** 2; // result is 25
// the same result as Math.pow(x,y)
let x = 5;
let z = Math.pow(x,2);

Comparison
Operator Description Example
== equal value let x = 5;
x == 8; // false
=== equal value and equal type let x = 5;
x === 5; // true
x === "5"; // false
!= not equal value let x = 5;
x != 8; // true
!== not equal value and equal type let x = 5;
x !== 5; // false
x !== "5"; // true
> greater than let x = 5;
x > 8; // false
< less than let x = 5;
x < 8; // true
>= greater than or equal value to let x = 5;
x >= 8; // false
x >= 5; // true
<= less than or equal value to let x = 5;
x <= 8; // true
x <= 5; // true
